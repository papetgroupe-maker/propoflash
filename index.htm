  const l=document.documentElement.lang;

  const left=Math.max(0,leftCount('free'));

  badge.textContent = l==='fr' ? `${left}/mois` : `${left}/mo`;

}


/* Send (simulate preview generation & count) */

$("#sendBrief").addEventListener("click",()=>{

  const t=I18N[document.documentElement.lang];

  const v=$("#brief").value.trim(); if(!v){$("#brief").focus();return;}

  if(PLAN==='free' && leftCount('free')<=0){

    alert(t.free_limit_reached);

    document.querySelector('#pricing').scrollIntoView({behavior:'smooth'}); return;

  }

  if(PLAN==='free'){ incCount('free'); updateFreeBadge(); }

  const left = PLAN==='free' ? leftCount('free') : '∞';

  alert( (document.documentElement.lang==='fr' ? "Brief reçu ✅\n\n" : "Brief received ✅\n\n")

    + v + "\n\n" + t.left_label + " : " + left);

});


/* 3D tilt + parallax */

(()=>{const el=$("#tilt");const rect=()=>el.getBoundingClientRect();

  const on=e=>{const r=rect(),x=(e.clientX-r.left)/r.width-.5,y=(e.clientY-r.top)/r.height-.5;

    el.style.transform=`rotateX(${(-y*3).toFixed(2)}deg) rotateY(${(x*3).toFixed(2)}deg)`};

  const off=()=>{el.style.transform="rotateX(0) rotateY(0)"}; el.addEventListener("mousemove",on); el.addEventListener("mouseleave",off);

})();

(()=>{const orb=$("#orb");window.addEventListener("mousemove",(e)=>{

  const x=(e.clientX/window.innerWidth-.5)*14,y=(e.clientY/window.innerHeight-.5)*14;

  orb.style.transform=`translate(${x}px, ${y}px)`;});})();
